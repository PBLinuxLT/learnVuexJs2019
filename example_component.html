# Example Vuex Component

<template>
  <div class="container">
    <slot><h3>Count = {{ getCount }} - {{ getName }} {{ job }} {{age}}</h3></slot>
    <button @click="addTen(10)">Add 10</button>
    <button @click="minusOne">Minus One</button>
    <button @click="uppercase">to Upper Case</button>
    <button @click="lowercase">to Lower Case</button>
  </div>
</template>

<script>
import { mapGetters, mapMutations, mapActions, mapState } from 'vuex'
export default {
  // name: 'ComponentName',
  data () {
    return {}
  },
  computed: {
    ...mapGetters('example', [
      'getCount',
      'getName'
    ]),
    ...mapState('example', [
      'job',
      'age'
    ])
  },
  methods: {
    ...mapMutations('example', [
      'addOne',
      'minusOne',
      'uppercase',
      'lowercase'
    ]),

    ...mapActions('example', [
      'addTen'
    ])
  }
}
</script>
