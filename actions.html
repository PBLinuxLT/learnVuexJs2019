# Vuex actions.js

// function (context, payload) {}
// context : commit, getters, dispatch
// or state or use destructuring as below.

export const addTen = ({ commit, dispatch }, payload) => {
  commit('add', payload)
  dispatch('cl')
}

export const cl = () => {
  console.log('Action cl dispatched from addTen')
}

Mutation add() which has state as a parameter and can receive a payload

export const add = (state, payload) => {
  state.count += payload
}
